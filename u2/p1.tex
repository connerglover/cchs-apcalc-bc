\begin{questions}

\noindent An object dropped from rest from the top of a tall building $y = 16t^2$ feet in the first $t$ seconds.

\item Find the average speed during the first 4 seconds of fall.\\[1.5cm]
\item Find the instantaneous speed of the object at $t = 4$ seconds and confirm your answer algebraically.\\[1.5cm]

\begin{multicols}{2}
\item Find $\lim_{x\to c} (2x^3-3x^2+x-1)$
\item Find $\lim_{x\to c} \frac{x^4 - x^3 - 1}{x^2 + 9}$
\end{multicols}

\vspace{1.5cm}

\noindent Determine the limit by substitution. Confirm your answer with a graph.

\begin{multicols}{2}
    \item $\lim_{x\to-4}(x+3)^{1998}$\\[3pt]\smallgraph
    \item $\lim_{x\to2} \frac{y^2 + 5y + 6}{y +2}$\\[3pt]\smallgraph
\end{multicols}

\vspace{1.5cm}

\begin{multicols}{2}
    \item $\lim_{x \to \frac{1}{2}}(\int x)$\\[3pt]\smallgraph
    \item $\lim_{x\to2} \sqrt{x + 3}$\\[3pt]\smallgraph
\end{multicols}
