\noindent An object dropped from rest from the top of a tall building $\displaystyle y = 16t^2$ feet in the first $\displaystyle t$ seconds.

\begin{enumerate}

\item Find the average speed during the first 4 seconds of fall.\\[2ex]
\item Find the instantaneous speed of the object at $\displaystyle t = 4$ seconds and confirm your answer algebraically.\\[2ex]

\begin{multicols}{2}
    \item Find $\displaystyle \lim_{x\to c} (2x^3-3x^2+x-1)$\\[2ex]
    \item Find $\displaystyle \lim_{x\to c} \frac{x^4 - x^3 - 1}{x^2 + 9}$\\[2ex]
\end{multicols}


\vspace{2ex}

\noindent Determine the limit by substitution. Confirm your answer with a graph.

\begin{multicols}{2}
    \item $\displaystyle \lim_{x\to-4}(x+3)^{1998}$\\[1ex]\smallgraph
    \item $\displaystyle \lim_{x\to2} \frac{x^2 + 5x + 6}{x + 2}$\\[1ex]\smallgraph
\end{multicols}


\vspace{2ex}

\begin{multicols}{2}
    \item $\displaystyle \lim_{x \to \frac{1}{2}}\lfloor x \rfloor$\\[1ex]\smallgraph
    \item $\displaystyle \lim_{x\to2} \sqrt{x + 3}$\\[1ex]\smallgraph
\end{multicols}

\noindent Explain why you cannot use substitution to determine the limit. Find the limit if it exists.

\begin{multicols}{2}
    \item $\displaystyle \lim_{x\to 0} \frac{1}{x^2}$\\[2ex]
    \item $\displaystyle \lim_{x\to 0} \frac{(4+x)^2 - 16}{x}$\\[2ex]
\end{multicols}

\vspace{2ex}

\noindent Determine the limit graphically. Confirm algebraically.

\begin{multicols}{2}
    \item $\displaystyle \lim_{t\to 2} \frac{t^2 - 3t + 2}{t^2 - 4}$\\[1ex]\smallgraph
    \item $\displaystyle \lim_{x\to 0} \frac{\frac{1}{2+x} - \frac{1}{2}}{x}$\\[1ex]\smallgraph
\end{multicols}

\vspace{2ex}

\begin{multicols}{2}
    \item $\displaystyle \lim_{x\to 0} \frac{\sin{2x}}{x}$\\[1ex]\smallgraph
    \item $\displaystyle \lim_{x\to 0} \frac{\sin{x}}{2x^2 - x}$\\[1ex]\smallgraph
\end{multicols}


\vspace{2ex}

\begin{multicols}{2}
    \item $\displaystyle \lim_{x\to 0} \frac{x + \sin{x}}{x}$\\[1ex]\smallgraph
    \item $\displaystyle \lim_{x\to 5} \frac{x^3 - 125}{x - 5}$\\[1ex]\smallgraph
\end{multicols}


\end{enumerate}
