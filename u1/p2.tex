\item Find the following, given $\displaystyle f(x) = x + 5$ and $\displaystyle g(x) = x^2 - 3$.

\begin{enumerate}[(a)]
\begin{multicols}{2}
    \item $\displaystyle f\!\left(g(x)\right)$\\[9ex]
    \item $\displaystyle g\!\left(f(x)\right)$\\[9ex]
    \item $\displaystyle f\!\left(g(0)\right)$\\[9ex]
    \item $\displaystyle g\!\left(f(0)\right)$\\[9ex]
    \item $\displaystyle g\!\left(g(-2)\right)$\\[9ex]
    \item $\displaystyle f\!\left(f(x)\right)$\\[9ex]
\end{multicols}
\end{enumerate}
