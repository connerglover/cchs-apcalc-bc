    \noindent Find the coordinate increments from $A$ to $B$.

    \begin{enumerate}

        \begin{multicols}{2}
            \item $\displaystyle A(-3, 2),\ B(-1, -2)$\\[3ex]
            \item $\displaystyle A(0, 4),\ B(0, -2)$\\[3ex]
        \end{multicols}

        \noindent Let $L$ be the line determined by points $A$ and $B$. Plot $A$ and $B$, find the slope of $L$, and graph $L$.

        \begin{multicols}{2}
            \item $\displaystyle A(-2, -1),\ B(1, -2)$\\[1ex]\smallgraph
            \item $\displaystyle A(2, 3),\ B(-1, 3)$\\[1ex]\smallgraph
        \end{multicols}

        \vspace{1ex}

        \noindent Write an equation for the vertical and horizontal line through the point $P$. Graph point $P$ and the lines.

        \begin{multicols}{2}
            \item $\displaystyle P\!\left(-1,\ \dfrac{4}{3}\right)$\\[1ex]\smallgraph
            \item $\displaystyle P\!\left(0,\ -\sqrt{2}\right)$\\[1ex]\smallgraph
        \end{multicols}

        \vspace{1ex}

        \noindent Write the point-slope equation for the line through point $P$ with slope $m$.

        \begin{multicols}{2}
            \item $\displaystyle P(-1, 1),\ m = -1$\\[4ex]
            \item $\displaystyle P(-4, 0),\ m = -2$\\[4ex]
        \end{multicols}

        \noindent Write the slope-intercept equation for the line with slope $m$ and $y$-intercept $b$.

        \begin{multicols}{2}
            \item $\displaystyle m = -1,\ b = -3$\\[4ex]
            \item $\displaystyle m = -2,\ b = -3$\\[4ex]
        \end{multicols}

        \noindent Write a general linear equation for the line through the two points.

        \begin{multicols}{2}
            \item $\displaystyle (-2, 0),\ (-2, -2)$\\[4ex]
            \item $\displaystyle (-5, 1),\ (2, -4)$\\[4ex]
        \end{multicols}

        \noindent Find the slope, the $y$-intercept, and graph the line.

        \begin{multicols}{2}
            \item $\displaystyle 3x + 4y = 12$\\[1ex]\smallgraph
            \item $\displaystyle \dfrac{x}{3} + \dfrac{y}{4} = 1$\\[1ex]\smallgraph
        \end{multicols}

        \vspace{1ex}

        \noindent Write an equation for the line through $P$ that is parallel to $L$ and perpendicular to $L$.

        \begin{multicols}{2}
            \item $\displaystyle P(-2, 2),\ L\colon 2x + y = 4$\\[6ex]
            \item $\displaystyle P\!\left(-1,\ \tfrac{1}{2}\right),\ L\colon y = 3$\\[6ex]
        \end{multicols}
