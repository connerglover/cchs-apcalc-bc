\noindent Solve for $x$:

\begin{align*}
    \ln x = 3t + 5 && e^{2x} = 10
\end{align*}

\noindent Properties of logarithms:

\begin{itemize}
    \item Product: $\log_b(mn) = \log_b m + \log_b n$
    \item Quotient: $\log_b \frac{m}{n} = \log_b m - \log_b n$
    \item Power: $\log_b m^p = p \cdot \log_b m$
\end{itemize}
