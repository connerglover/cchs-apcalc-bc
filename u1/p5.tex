\noindent Graph with a calculator and determine if the function is one-to-one.

\begin{enumerate}

    \begin{multicols}{2}
        \item $\displaystyle y = 2|x|$\\[1ex]\smallgraph
        \item $\displaystyle f(x) = \dfrac{1}{x}$\\[1ex]\smallgraph
    \end{multicols}

    \begin{multicols}{2}
        \item $\displaystyle y = \dfrac{1}{x^2 + 1}$\\[1ex]\smallgraph
        \item $\displaystyle y = \lfloor\lfloor x \rfloor\rfloor$\\[1ex]\smallgraph
    \end{multicols}

    \vspace{1ex}

    \noindent Determine whether the function is a one-to-one function and has an inverse function.

    \begin{multicols}{2}
        \item $\displaystyle y = x^2 + 5x$\\[5ex]
        \item $\displaystyle y = x^3 + x$\\[5ex]
    \end{multicols}

    \begin{multicols}{2}
        \item $\displaystyle y = \ln(x^2)$\\[5ex]
        \item $\displaystyle y = 2^{3-x}$\\[5ex]
    \end{multicols}

    \noindent Find $\displaystyle f^{-1}$ and verify that $\displaystyle (f \circ f^{-1})(x) = (f^{-1} \circ f)(x) = x$.

    \begin{multicols}{2}
        \item $\displaystyle f(x) = 5 - 4x$\\[6ex]
        \item $\displaystyle f(x) = x^2 + 1,\quad x \ge 0$\\[6ex]
    \end{multicols}

    \begin{multicols}{2}
        \item $\displaystyle f(x) = x^{2/3},\quad x \ge 0$\\[6ex]
        \item $\displaystyle f(x) = x^2 + 2x + 1,\quad x \ge -1$\\[6ex]
    \end{multicols}

    \begin{multicols}{2}
        \item $\displaystyle f(x) = \dfrac{1}{x^3}$\\[6ex]
        \item $\displaystyle f(x) = \dfrac{x + 3}{x - 2}$\\[6ex]
    \end{multicols}

    \newpage

    \noindent Graph $f$, $f^{-1}$, and $y = x$ parametrically.

    \begin{multicols}{2}
        \item $\displaystyle f(x) = 3^x$\\[1ex]\smallgraph
        \item $\displaystyle f(x) = \ln x$\\[1ex]\smallgraph
    \end{multicols}

    \begin{multicols}{2}
        \item $\displaystyle f(x) = \log x$\\[1ex]\smallgraph
        \item $\displaystyle f(x) = \sin^{-1} x$\\[1ex]\smallgraph
    \end{multicols}

    \vspace{1ex}

    \noindent Solve the equation algebraically. Support your solution graphically.

    \begin{multicols}{2}
        \item $\displaystyle e^{0.05t} = 3$\\[5ex]
        \item $\displaystyle 2^x + 2^{-x} = 5$\\[5ex]
    \end{multicols}

    \noindent Solve for $y$.

    \begin{multicols}{2}
        \item $\displaystyle \ln y = 2t + 4$\\[5ex]
        \item $\displaystyle \ln(y - 1) - \ln 2 = x + \ln x$\\[5ex]
    \end{multicols}

    \noindent Draw the graph and determine the domain and range.

    \begin{multicols}{2}
        \item $\displaystyle y = -3\log(x + 2) + 1$\\[1ex]\smallgraph
        \item $\displaystyle y = \log_3(x - 4)$\\[1ex]\smallgraph
    \end{multicols}

    \newpage

    \item Find a formula for $\displaystyle f^{-1}$ and verify that $\displaystyle (f \circ f^{-1})(x) = (f^{-1} \circ f)(x) = x$.
    \[f(x) = \dfrac{100}{1 + 2^{-x}}\]

    \vspace{33ex}

    \item Prove that the function $f$ is its own inverse.

    \begin{enumerate}[(a)]
        \item $\displaystyle f(x) = \sqrt{1 - x^2}$\\[6ex]
        \item $\displaystyle f(x) = \dfrac{1}{x}$\\[6ex]
    \end{enumerate}

\end{enumerate}
